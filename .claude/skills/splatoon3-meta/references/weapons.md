## Programmatic Access

```python
from splatnlp.dashboard.utils.converters import get_weapon_properties_df
from splatnlp.preprocessing.transform.mappings import generate_maps

# Get full weapon DataFrame
df = get_weapon_properties_df()  # Returns weapon_id, sub, special, class

# Get ID to name mapping
_, id_to_name, _ = generate_maps()
name = id_to_name.get("1111")  # "Octobrush Nouveau"

# In mechinterp context
from splatnlp.mechinterp.skill_helpers import load_context
ctx = load_context("ultra")
name = ctx.id_to_weapon_display_name(8)  # "Octobrush Nouveau"
```

## Kit Lookup from MechInterp Token IDs

**IMPORTANT**: MechInterp uses sequential token IDs (0, 1, 2, ...) but weapon properties use game IDs ("1111", "220", ...). Chain through `inv_weapon_vocab` to convert:

```python
from splatnlp.mechinterp.skill_helpers import load_context
from splatnlp.dashboard.utils.converters import (
    get_weapon_properties,
    get_sub_type_translation,
    get_special_type_translation,
)

ctx = load_context("ultra")
props = get_weapon_properties()
subs_t = get_sub_type_translation()
specials_t = get_special_type_translation()

def get_kit_from_token_id(token_id: int) -> tuple[str, str]:
    """Convert mechinterp token_id to (sub_name, special_name)."""
    raw = ctx.inv_weapon_vocab.get(token_id, "")
    if raw.startswith("weapon_id_"):
        numeric_id = raw.split("_")[-1]  # "weapon_id_1111" -> "1111"
        if numeric_id in props:
            p = props[numeric_id]
            sub = subs_t.get(p['sub'], p['sub'])
            special = specials_t.get(p['special'], p['special'])
            return sub, special
    return "?", "?"

# Example usage
token_id = 8  # From mechinterp weapon_sweep
sub, special = get_kit_from_token_id(token_id)
print(f"Token {token_id}: {ctx.id_to_weapon_display_name(token_id)} - {sub} + {special}")
# Output: Token 8: Octobrush Nouveau - Squid Beakon + Ink Storm
```

This pattern is essential for kit-based feature analysis when examining weapon_sweep results.
